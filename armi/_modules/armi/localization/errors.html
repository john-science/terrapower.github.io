[docs]def general_single_TooManyOrTooFewMatchesFor(matchedObjects):
    return Exception(
        "Expected single match, but got {} matches:\n  {}".format(
            len(matchedObjects), "\n  ".join(str(mo) for mo in matchedObjects)
        )
    )

        )
